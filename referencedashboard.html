""" NIFTY Options — Depth / Order Book Analyzer v2 ------------------------------------------------ Adds: IQR stats, sweep slippage (first vs last price), weighted avg slippage, and redesigned dashboard. """ import os, json, re, math, glob, threading, webbrowser, http.server, socketserver import pandas as pd import numpy as np # ───────────────────────────────────────── # USER CONFIG # ───────────────────────────────────────── FOLDER = "/Users/urmila/fo_2026-05-27" LOT_SIZE = 65 ORDER_LOTS = 100 PORT = 8766 DOWNSAMPLE = 20 KEY_DEPTHS = [1, 2, 3, 4, 5, 10, 15, 20, 30] # ───────────────────────────────────────── try: SCRIPT_DIR = os.path.dirname(os.path.abspath(__file__)) except NameError: SCRIPT_DIR = os.getcwd() OUTPUT_JSON = os.path.join(SCRIPT_DIR, "depth_results.json") OUTPUT_HTML = os.path.join(SCRIPT_DIR, "depth_dashboard.html") FILE_RE = re.compile( r"(\d{4}-\d{2}-\d{2})_NIFTY_(\d+)_(CE|PE)_(.+?)_NSE_FO_(\d+)_depth_levels\.csv" ) # ── ATM ──────────────────────────────────────────────────── def get_atm(spot, interval=50): lower = math.floor(spot / interval) * interval mid = lower + interval / 2 return lower if spot <= mid else lower + interval # ── Scan folder ──────────────────────────────────────────── def scan_folder(folder): files=sorted(glob.glob(os.path.join(folder, "*_depth_levels.csv" ))) strikes=[] for f in files: m=FILE_RE.match(os.path.basename(f)) if m: strikes.append({ "path" : f, "date" : m.group(1), "strike" : int(m.group(2)), "type" : m.group(3), "expiry" : m.group(4).replace("_", " " ), "label" : f"NIFTY {m.group(2)} {m.group(3)} ({m.group(4).replace('_',' ')})", "key" : f"{m.group(2)}_{m.group(3)}", }) return sorted(strikes, key=lambda x: (x["strike"], x["type"])) # ── Load & pivot ─────────────────────────────────────────── def load_and_pivot(path): df = pd.read_csv(path, on_bad_lines="skip") df["ts"] = pd.to_datetime( df["received_at_ist"].str.replace(" IST", "", regex=False), errors="coerce" ) df = df.dropna(subset=["ts"]) df["time_str"] = df["ts"].dt.strftime("%H:%M:%S") snapshots = [] for _, group in df.groupby("feed_timestamp_ms", sort=True): g = group.set_index("level") snapshots.append({ "ts" : group["time_str"].iloc[0], "ltp" : float(group["last_traded_price"].iloc[0]), "bid_qty": [int(g.loc[l, "bid_quantity"]) if l in g.index else 0 for l in range(1, 31)], "bid_px" : [float(g.loc[l, "bid_price"]) if l in g.index else 0 for l in range(1, 31)], "ask_qty": [int(g.loc[l, "ask_quantity"]) if l in g.index else 0 for l in range(1, 31)], "ask_px" : [float(g.loc[l, "ask_price"]) if l in g.index else 0 for l in range(1, 31)], }) return snapshots # ── Fill simulation — returns BOTH slippage types ────────── def fill_per_level(snap, side, qty, levels): """ Walk the specified levels (sorted). Returns: - weighted_slip : avg_fill_price - L1_price (true cost impact) - sweep_slip : last_fill_price - L1_price (how deep book was swept) """ prices = snap["ask_px"] if side == "buy" else snap["bid_px"] qtys = snap["ask_qty"] if side == "buy" else snap["bid_qty"] ref = prices[0] if prices else 0 remaining = qty cost = 0.0 min_depth = None last_px = ref breakdown = [] for l in sorted(levels): if remaining <= 0: break idx = l - 1 px = prices[idx] if idx < len(prices) else 0 q = qtys[idx] if idx < len(qtys) else 0 take = min(remaining, q) if take > 0: cost += px * take last_px = px remaining -= take breakdown.append({ "level" : l, "price" : round(px, 2), "avail_qty": int(q), "taken_qty": int(take), }) if remaining <= 0 and min_depth is None: min_depth = l fill_qty = qty - remaining avg_price = round(cost / fill_qty, 4) if fill_qty > 0 else 0 # Weighted slippage: avg fill price vs L1 weighted_slip = round(avg_price - ref, 4) if side == "buy" else round(ref - avg_price, 4) # Sweep slippage: last touched price vs L1 sweep_slip = round(last_px - ref, 4) if side == "buy" else round(ref - last_px, 4) return { "fill_qty" : int(fill_qty), "unfilled" : int(remaining), "filled" : remaining <= 0, "avg_price" : avg_price, "ref_price" : round(ref, 4), "last_px" : round(last_px, 4), "weighted_slip": weighted_slip, "sweep_slip" : sweep_slip, "min_depth" : min_depth, "breakdown" : breakdown, } # ── IQR aggregation (replaces simple avg/min/max) ────────── def iqr_stats(arr): """ Returns median, IQR, outlier fences, plus legacy avg/min/max. Outlier fence: Q1 - 1.5*IQR to Q3 + 1.5*IQR """ a = [x for x in arr if x is not None and not math.isnan(x)] if not a: return { "avg": None, "min": None, "max": None, "median": None, "q1": None, "q3": None, "iqr": None, "fence_lo": None, "fence_hi": None, "outlier_pct": None, } q1 = float(np.percentile(a, 25)) med = float(np.percentile(a, 50)) q3 = float(np.percentile(a, 75)) iqr = q3 - q1 fence_lo = q1 - 1.5 * iqr fence_hi = q3 + 1.5 * iqr outliers = [x for x in a if x < fence_lo or x > fence_hi] return { "avg" : round(float(np.mean(a)), 4), "min" : round(float(np.min(a)), 4), "max" : round(float(np.max(a)), 4), "median" : round(med, 4), "q1" : round(q1, 4), "q3" : round(q3, 4), "iqr" : round(iqr, 4), "fence_lo" : round(fence_lo, 4), "fence_hi" : round(fence_hi, 4), "outlier_pct": round(len(outliers) / len(a) * 100, 1), } # ── Analyze one strike ───────────────────────────────────── def analyze(path, lot_size=LOT_SIZE, order_lots=ORDER_LOTS): print(f" Loading {os.path.basename(path)} ...") snapshots = load_and_pivot(path) total = len(snapshots) sampled = snapshots[::DOWNSAMPLE] order_qty = order_lots * lot_size ts_labels = [] ltp_series = [] spread_series = [] per_level = { "buy" : {d: {"qty": [], "fill_qty": [], "w_slip": [], "s_slip": [], "avg_px": []} for d in KEY_DEPTHS}, "sell": {d: {"qty": [], "fill_qty": [], "w_slip": [], "s_slip": [], "avg_px": []} for d in KEY_DEPTHS}, } cum_bid = {d: [] for d in KEY_DEPTHS} cum_ask = {d: [] for d in KEY_DEPTHS} min_depth_buy_series = [] min_depth_sell_series = [] compact_snaps = [] for snap in sampled: ts = snap["ts"] ltp = snap["ltp"] ts_labels.append(ts) ltp_series.append(ltp) spread = round(snap["ask_px"][0] - snap["bid_px"][0], 4) spread_series.append(spread) compact_snaps.append({ "ts" : ts, "ltp" : ltp, "bp" : [round(p, 2) for p in snap["bid_px"]], "bq" : snap["bid_qty"], "ap" : [round(p, 2) for p in snap["ask_px"]], "aq" : snap["ask_qty"], "spread": spread, }) for d in KEY_DEPTHS: bid_q = sum(snap["bid_qty"][:d]) ask_q = sum(snap["ask_qty"][:d]) cum_bid[d].append(round(bid_q / lot_size, 2)) cum_ask[d].append(round(ask_q / lot_size, 2)) for side in ["buy", "sell"]: res = fill_per_level(snap, side, order_qty, list(range(1, d+1))) per_level[side][d]["qty"].append( round((ask_q if side == "buy" else bid_q) / lot_size, 2) ) per_level[side][d]["fill_qty"].append(round(res["fill_qty"] / lot_size, 2)) per_level[side][d]["w_slip"].append(res["weighted_slip"]) per_level[side][d]["s_slip"].append(res["sweep_slip"]) per_level[side][d]["avg_px"].append(res["avg_price"]) bf30 = fill_per_level(snap, "buy", order_qty, list(range(1, 31))) sf30 = fill_per_level(snap, "sell", order_qty, list(range(1, 31))) min_depth_buy_series.append(bf30["min_depth"]) min_depth_sell_series.append(sf30["min_depth"]) # ── All-day IQR aggregates per depth ── allday = {"buy": {}, "sell": {}} for side in ["buy", "sell"]: for d in KEY_DEPTHS: allday[side][str(d)] = { "qty" : iqr_stats(per_level[side][d]["qty"]), "fill" : iqr_stats(per_level[side][d]["fill_qty"]), "w_slip": iqr_stats(per_level[side][d]["w_slip"]), "s_slip": iqr_stats(per_level[side][d]["s_slip"]), } allday["min_depth_buy"] = iqr_stats([x for x in min_depth_buy_series if x]) allday["min_depth_sell"] = iqr_stats([x for x in min_depth_sell_series if x]) allday["spread"] = iqr_stats(spread_series) # ── Latest snapshot slippage curve ── latest = snapshots[-1] bid_depth_slip = [round(latest["bid_px"][0] - latest["bid_px"][i], 4) for i in range(30)] ask_depth_slip = [round(latest["ask_px"][i] - latest["ask_px"][0], 4) for i in range(30)] s = allday["spread"] summary = { "total_snapshots" : total, "sampled_snapshots": len(sampled), "time_start" : ts_labels[0] if ts_labels else "", "time_end" : ts_labels[-1] if ts_labels else "", "ltp_latest" : latest["ltp"], "spread_median" : s["median"], "spread_q1" : s["q1"], "spread_q3" : s["q3"], "spread_iqr" : s["iqr"], "spread_fence_hi" : s["fence_hi"], "spread_outlier_pct": s["outlier_pct"], "spread_avg" : s["avg"], "spread_min" : s["min"], "spread_max" : s["max"], "lot_size" : lot_size, "default_lots" : order_lots, "default_qty" : order_qty, "ltp_min" : round(float(min(ltp_series)), 2) if ltp_series else 0, "ltp_max" : round(float(max(ltp_series)), 2) if ltp_series else 0, "ltp_avg" : round(float(np.mean(ltp_series)), 2) if ltp_series else 0, } return { "summary" : summary, "key_depths" : KEY_DEPTHS, "allday" : allday, "snapshots" : compact_snaps, "ts_labels" : ts_labels, "ltp_series" : ltp_series, "spread_series" : spread_series, "cum_bid" : {str(k): v for k, v in cum_bid.items()}, "cum_ask" : {str(k): v for k, v in cum_ask.items()}, "per_level_buy" : {str(d): per_level["buy"][d] for d in KEY_DEPTHS}, "per_level_sell" : {str(d): per_level["sell"][d] for d in KEY_DEPTHS}, "min_depth_buy" : min_depth_buy_series, "min_depth_sell" : min_depth_sell_series, "bid_depth_slip" : bid_depth_slip, "ask_depth_slip" : ask_depth_slip, } # ── Main ─────────────────────────────────────────────────── def run_analysis(): print(f"Scanning {FOLDER} ...") strikes = scan_folder(FOLDER) if not strikes: print("No depth CSV files found."); return print(f"Found {len(strikes)} files:") for s in strikes: print(f" {s[' label']}") results={} for s in strikes: try: data=analyze(s["path"], LOT_SIZE, ORDER_LOTS) data["meta"]=s results[s["key"]]=data except Exception as e: print(f" Skipping {s['label']}: {e}") output={"strikes": strikes, "results" : results, "lot_size" : LOT_SIZE, "default_lots" : ORDER_LOTS} with open(OUTPUT_JSON, "w" ) as f: json.dump(output, f) print(f"\nDone → {OUTPUT_JSON} ({len(strikes)} strikes)") # """ # ── Serve ────────────────────────────────────────────────── class ThreadedHTTPServer(socketserver.ThreadingMixIn, socketserver.TCPServer): allow_reuse_address = True def serve_and_open(): os.chdir(SCRIPT_DIR) class Q(http.server.SimpleHTTPRequestHandler): def log_message(self,*a): pass def handle_error(self,*a): pass with ThreadedHTTPServer(("", PORT), Q) as s: url = f"http://localhost:{PORT}/depth_dashboard.html" print(f"\n Dashboard → {url}") print(" Press Ctrl+C to stop\n") threading.Timer(1.0, lambda: webbrowser.open(url)).start() s.serve_forever() if __name__ == "__main__": run_analysis() with open(OUTPUT_HTML, "w") as f: f.write(HTML) print(f" Dashboard written → {OUTPUT_HTML}") serve_and_open()
